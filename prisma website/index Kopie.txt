    function nextStep(step, lang) {
      if (lang) selectedLanguage = lang; // Set selected language if provided
      if (step === "start") {
        location.reload(); // Refresh the page
        return;
      }
      currentStep = step;
      renderStep();
    }

    function submitName() {
      const nameInput = document.getElementById('name').value.trim();
      if (nameInput) {
        userName = nameInput;
        nextStep('askVisited');
      }
    }

    function renderStep() {
      const stepData = translations[selectedLanguage][currentStep];
      const question = typeof stepData.question === "function" ? stepData.question(userName) : stepData.question;

      document.getElementById('question').textContent = question;
      const buttonsContainer = document.getElementById('buttons');
      const inputSection = document.getElementById('input-section');
      const imageElement = document.getElementById('image');

      buttonsContainer.innerHTML = ''; // Clear previous buttons
      inputSection.style.display = stepData.input ? 'block' : 'none'; // Show/hide input
      imageElement.style.display = stepData.image ? 'block' : 'none'; // Show/hide image

      if (stepData.image) imageElement.src = stepData.image;

      if (stepData.buttons) {
        stepData.buttons.forEach(button => {
          const btn = document.createElement('button');
          btn.textContent = button.text;
          btn.onclick = () => nextStep(button.nextStep, button.lang);
          buttonsContainer.appendChild(btn);
        });
      }
    }

    renderStep(); // Initialize the first step
